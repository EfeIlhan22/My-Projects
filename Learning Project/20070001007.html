def sigmoid(X):
    return 1/(1+np.exp(-X))

def sigmoid_deriv(X):
    return sigmoid(X)*(1-sigmoid(X))

def forward(X,W1,W2,predict=False):
    a1=np.matmul(X,W1) #matrix çarpımı yaptım
    Z1=sigmoid(a1)

    bias=np.ones((len(Z1),1))
    Z1=np.concatenate((bias,Z1),axis=1)  #biası z1 in bir parçası gibi düşünüp formülden çıkardım 
    a2=np.matmul(Z1,W2)
    Z2=sigmoid(a2)
    if predict:
        return Z2
    return a1,Z1,a2,Z2

# Implementing the Logistic Regression
class TwoLayerMLP:
    def __init__(self, input_size, hidden_size, output_size,batch_size, reg_lambda,rho=0.9,learning_rate=0.01):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        self.reg_lambda = reg_lambda
        self.rho=rho
        self.batch_size=batch_size
        self.init_weights()
        self.zero_grad()
        self.learning_rate = learning_rate

    def init_weights(self):
        self.W1 = np.random.randn(self.input_size, self.hidden_size) * 0.01
        self.b1 = np.zeros((1, self.hidden_size))
        self.W2 = np.random.randn(self.hidden_size, self.output_size) * 0.01
        self.b2 = np.zeros((1, self.output_size))
        self.velocity_W1 = np.zeros_like(self.W1)
        self.velocity_b1 = np.zeros_like(self.b1)
        self.velocity_W2 = np.zeros_like(self.W2)
        self.velocity_b2 = np.zeros_like(self.b2)


    def zero_grad(self):
        self.dW1 = np.zeros_like(self.W1)
        self.db1 = np.zeros_like(self.b1)
        self.dW2 = np.zeros_like(self.W2)
        self.db2 = np.zeros_like(self.b2)


    def update_weights(self, learning_rate):
           # Update velocity terms
        self.velocity_W1 = self.rho * self.velocity_W1 + self.dW1
        self.velocity_b1 = self.rho * self.velocity_b1 + self.db1
        self.velocity_W2 = self.rho * self.velocity_W2 + self.dW2
        self.velocity_b2 = self.rho * self.velocity_b2 + self.db2

        # Update parameters
        self.W1 -= learning_rate * self.velocity_W1
        self.b1 -= learning_rate * self.velocity_b1
        self.W2 -= learning_rate * self.velocity_W2
        self.b2 -= learning_rate * self.velocity_b2

      

    def sigmoid(self, Z):
        return 1 / (1 + np.exp(-Z))

    def sigmoid_derivative(self, Z):
        s = self.sigmoid(Z)
        return s * (1 - s)

    def forward(self, X, Y=None):
        self.X = X
        if Y is not None:
            self.Y = Y
            
        
        self.Z1=self.sigmoid(np.dot(X,self.W1)+self.b1)
        self.H1=self.Z1
        self.Z2=self.sigmoid(np.dot(self.Z1,self.W2)+self.b2)
        self.H2=self.Z2
        
        return self.H2

    def compute_loss(self, Y_pred=None, Y=None):
        if Y_pred is None:
            Y_pred = self.H2  
        if Y is None:
            Y = self.Y
            

        batch_size = Y.shape[0]
        data_loss = np.sum((Y_pred - Y) ** 2) / batch_size
        reg_loss = (self.reg_lambda / 2) *np.sum(np.square(self.W1)+np.sum(np.square(self.W2))) / batch_size
        loss = data_loss + reg_loss
        return loss
    

    def backward(self,X,Y):
        Y_pred = self.H2 
        batch_size = Y.shape[0]

        dZ2 = (Y_pred - Y) * self.sigmoid_derivative(self.Z2)
        self.dW2 = (np.dot(self.H1.T, dZ2) + self.reg_lambda * self.W2) / batch_size
        self.db2 = np.sum(dZ2, axis=0, keepdims=True) / batch_size
        #dZ1 = np.dot(dZ2[:, :self.hidden_size], self.W2.T) * self.sigmoid_derivative(self.Z1)
        dZ1 = np.dot(dZ2, self.W2.T) * self.sigmoid_derivative(self.Z1)
        self.dW1 = (np.dot(X.T, dZ1) + self.reg_lambda * self.W1) / batch_size
        self.db1 = np.sum(dZ1, axis=0, keepdims=True) / batch_size

    def predict(self, X):
        # Predict output for new data
        return self.forward(X)
    def train(self, X, y, epochs=1000):
        for i in range(epochs):
            # Forward propagation
            self.forward(X)
            
            # Backpropagation
            self.backward(X, y)   
            self.update_weights(self.learning_rate)

class SGD:
    def __init__(self, model, learning_rate):
        self.model = model
        self.learning_rate = learning_rate
     
    def step(self):
        self.model.update_weights(self.learning_rate)
        
    def zero_grad(self):
        self.model.zero_grad()

def train_sgd(model, optimizer, X_train, Y_train, epochs, batch_size):
    loss_list = []
    n_samples = X_train.shape[0]

    for epoch in range(epochs):
        total_loss = 0
        indices = np.random.permutation(n_samples)

        for i in range(0, n_samples, batch_size):
            batch_indices = indices[i:i+batch_size]
            X_batch = X_train[batch_indices]
            Y_batch = Y_train[batch_indices]

            model.forward(X_batch, Y_batch)  
            loss = model.compute_loss()  
            model.backward(X_batch, Y_batch)
            optimizer.step()
            optimizer.zero_grad()
            total_loss += loss

        avg_loss = total_loss / (n_samples / batch_size)
        loss_list.append(avg_loss)
            
    return loss_list

R2 (Coefficient of Determination)¶

The coefficient of determination, denoted as R2, is a statistical measure that represents the proportion of the variance for a dependent variable that's explained by an independent variable(s) in a regression model. It is calculated as:

R^2 = 1 - \frac{SS_{res}}{SS_{tot}}

Where:

  • ( SS_{res} ) is the sum of squares of residuals (or errors)
  • ( SS_{tot} ) is the total sum of squares

R2 ranges from 0 to 1, where:

  • 0 indicates that the model does not explain any of the variance in the target variable.
  • 1 indicates that the model explains all the variance in the target variable.

RMSE (Root Mean Square Error)¶

The root mean square error, denoted as RMSE, measures the average magnitude of the errors between predicted and actual values in a regression problem. It is calculated as the square root of the average of squared differences between predicted and actual values:

[ RMSE = \sqrt{\frac{1}{n}\sum_{i=1}^{n}(y_{i} - \hat{y}_{i})^2} ]

Where:

  • ( n ) is the number of data points
  • ( y_{i} ) is the actual value
  • ( \hat{y}_{i} ) is the predicted value

RMSE can range from 0 to positive infinity. Lower RMSE values indicate better model performance, where an RMSE of 0 would denote a perfect prediction.

Range and Interpretation¶

  • R2 Range: 0 to 1

  • RMSE Range: 0 to positive infinity

  • Perfect Prediction:

    • R2: 1
    • RMSE: 0

    Reasoning:

    • A perfect R2 score of 1 means that the model explains all the variability of the response data around its mean.
    • A RMSE of 0 indicates that the predicted values perfectly match the observed values, meaning there are no errors.

  • Terrible Prediction:

    • R2: Close to 0
    • RMSE: High value (much greater than 0)

    Reasoning:

    • A low R2 score close to 0 suggests that the model does not explain much of the variability in the response data.
    • A high RMSE value indicates large discrepancies between predicted and actual values, implying poor model performance.
