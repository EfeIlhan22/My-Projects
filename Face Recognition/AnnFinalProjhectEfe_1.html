def cosine_similarity(embedding1, embedding2):
    return F.cosine_similarity(embedding1.unsqueeze(0), embedding2.unsqueeze(0))

def find_most_similar(embedding, gallery_embeddings):
    max_similarity = -1
    most_similar_person = None
    for person, embeddings in gallery_embeddings.items():
        for gallery_embedding in embeddings:
            similarity = cosine_similarity(embedding, gallery_embedding).item()
            if similarity > max_similarity:
                max_similarity = similarity
                most_similar_person = person
    return most_similar_person, max_similarity

def add_new_person_to_gallery(person_name, embedding, gallery_embeddings, gallery_path):
    if person_name not in gallery_embeddings:
        gallery_embeddings[person_name] = []
    gallery_embeddings[person_name].append(embedding)
    save_gallery_to_json(gallery_embeddings, gallery_path)

import torch
import torch.nn as nn
import torch.nn.functional as F
from torchvision import models

class SimpleMultiLabelModel(nn.Module):
    def __init__(self, num_classes_identity, num_classes_style):
        super(SimpleMultiLabelModel, self).__init__()
        self.backbone = models.mobilenet_v2(pretrained=True)
        for param in self.backbone.parameters():
            param.requires_grad = False  # Freeze the backbone
        num_features = self.backbone.classifier[1].in_features
        self.backbone.classifier = nn.Identity()
        self.fc_identity = nn.Sequential(
            nn.Linear(num_features, 256),
            nn.ReLU(),
            nn.Linear(256, num_classes_identity)
        )
        self.fc_style = nn.Sequential(
            nn.Linear(num_features, 256),
            nn.ReLU(),
            nn.Linear(256, num_classes_style)
        )

    def forward(self, x):
        features = self.backbone(x)
        identity_output = self.fc_identity(features)
        style_output = self.fc_style(features)
        return identity_output, style_output, features

This code defines a simple multi-label classification model using PyTorch. The model is built on a pre-trained MobileNetV2 backbone, which is frozen to prevent updates during training. The backbone's classifier is replaced with two separate fully connected layers for identity and style classification.¶

MobileNetV2 is used for this problem because it provides a good balance between accuracy and computational efficiency. Here are the key reasons for its selection:¶

  • Lightweight Architecture: MobileNetV2 is designed to be computationally efficient with fewer parameters and operations compared to many other deep learning models. This makes it suitable for deployment on resource-constrained devices like mobile phones and embedded systems.

  • Pre-trained Weights: Using a pre-trained MobileNetV2 allows leveraging the features learned from a large dataset (like ImageNet), which can improve the model's performance and accelerate convergence, especially useful when labeled data is limited.

  • Feature Extraction: MobileNetV2 is effective at extracting useful features from images, which can then be used for multiple tasks (such as identity and style classification in this case).

  • Versatility: The architecture is flexible and can be easily adapted for various applications, including multi-label classification tasks as demonstrated by using separate fully connected layers for identity and style outputs.

def train_model(model, train_loader, criterion_person, criterion_style, optimizer, num_epochs):
    model.train()
    for epoch in range(num_epochs):
        running_loss_person = 0.0
        running_loss_style = 0.0
        correct_person = 0
        correct_style = 0
        total = 0

        for inputs, person_labels, style_labels in train_loader:
            inputs, person_labels, style_labels = inputs.to(device), person_labels.to(device), style_labels.to(device)
            optimizer.zero_grad()

            person_outputs, style_outputs,_ = model(inputs)
            loss_person = criterion_person(person_outputs, person_labels)
            loss_style = criterion_style(style_outputs, style_labels)
            loss = loss_person + loss_style
            loss.backward()
            optimizer.step()

            running_loss_person += loss_person.item()
            running_loss_style += loss_style.item()
            _, predicted_person = torch.max(person_outputs, 1)
            _, predicted_style = torch.max(style_outputs, 1)
            correct_person += (predicted_person == person_labels).sum().item()
            correct_style += (predicted_style == style_labels).sum().item()
            total += person_labels.size(0)

        epoch_loss_person = running_loss_person / total
        epoch_loss_style = running_loss_style / total
        epoch_acc_person = correct_person / total
        epoch_acc_style = correct_style / total

        print(f'Epoch [{epoch+1}/{num_epochs}], Person Loss: {epoch_loss_person:.4f}, Style Loss: {epoch_loss_style:.4f}, Person Acc: {epoch_acc_person:.4f}, Style Acc: {epoch_acc_style:.4f}')

This function train_model is designed to train the SimpleMultiLabelModel for a specified number of epochs. The model is trained on both person and style classifications using separate loss functions for each.¶

def create_embedding_for_new_person(model, new_data, device):
    model.eval()
    new_data = new_data.to(device)
    with torch.no_grad():
        _, _, person_features = model(new_data)
    return person_features.cpu()

def match_new_person_with_gallery(new_embedding, gallery_embeddings):
    gallery_tensors = torch.stack([torch.mean(torch.stack(embeds), dim=0) for embeds in gallery_embeddings.values()])
    cos_sim = torch.matmul(new_embedding, gallery_tensors.T)
    cos_sim = cos_sim / (torch.norm(new_embedding, dim=1, keepdim=True) * torch.norm(gallery_tensors, dim=1))
    return cos_sim

def add_new_person_to_gallery(new_person_name, new_embedding, gallery_embeddings):
    if new_person_name not in gallery_embeddings:
        gallery_embeddings[new_person_name] = []
    gallery_embeddings[new_person_name].append(new_embedding)
    return gallery_embeddings

def evaluate_model(model, val_dataset, gallery_json_path, device):
    # Load gallery embeddings from JSON file
    loaded_gallery_embeddings = read_gallery_from_json(gallery_json_path)
    print(f"Loaded gallery embeddings: {len(loaded_gallery_embeddings)} persons")

    # Create embeddings for the probe set (validation set)
    probe_embeddings = create_gallery_embeddings(model, val_dataset, device)
    print(f"Created probe embeddings: {len(probe_embeddings)} persons")

    # Cosine similarity for gallery-probe matching
    gallery_tensors = torch.stack([torch.mean(torch.stack(embeds), dim=0) for embeds in loaded_gallery_embeddings.values()])
    probe_tensors = torch.stack([torch.mean(torch.stack(embeds), dim=0) for embeds in probe_embeddings.values()])

    cos_sim = torch.matmul(probe_tensors, gallery_tensors.T)
    cos_sim = cos_sim / (torch.norm(probe_tensors, dim=1, keepdim=True) * torch.norm(gallery_tensors, dim=1))

    # Create a mapping from person names to indices
    person_to_idx = {person: idx for idx, person in enumerate(loaded_gallery_embeddings.keys())}
    idx_to_person = {idx: person for person, idx in person_to_idx.items()}

    # Add an "unknown" category
    unknown_idx = len(person_to_idx)
    idx_to_person[unknown_idx] = "unknown"

    # Convert predicted labels to indices
    predicted_indices = torch.argmax(cos_sim, dim=1).cpu().numpy()

    # True indices based on the probe set
    true_indices = []
    for person_name, embeds in probe_embeddings.items():
        person_name = str(person_name)
        if person_name in person_to_idx:
            person_idx = person_to_idx[person_name]
        else:
            person_idx = unknown_idx
        true_indices.extend([person_idx] * len(embeds))

    if len(predicted_indices) != len(true_indices):
        print(f"Warning: Length mismatch - Predicted: {len(predicted_indices)}, True: {len(true_indices)}")
        min_length = min(len(predicted_indices), len(true_indices))
        predicted_indices = predicted_indices[:min_length]
        true_indices = true_indices[:min_length]

    accuracy = (torch.tensor(predicted_indices) == torch.tensor(true_indices)).float().mean().item()
    print(f'Validation Set Accuracy: {accuracy:.4f}')

    # Visualization
    # Convert indices to person names
    true_labels = [idx_to_person[idx] for idx in true_indices]
    predicted_labels = [idx_to_person[idx] for idx in predicted_indices]

    # Calculate confusion matrix
    cm = confusion_matrix(true_labels, predicted_labels, labels=list(idx_to_person.values()))

    # Plot confusion matrix
    plt.figure(figsize=(10, 8))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', xticklabels=idx_to_person.values(), yticklabels=idx_to_person.values())
    plt.xlabel('Predicted Labels')
    plt.ylabel('True Labels')
    plt.title('Confusion Matrix')
    plt.show()

    # Plot bar chart for true vs predicted labels
    plt.figure(figsize=(12, 6))
    true_count = pd.Series(true_labels).value_counts().sort_index()
    pred_count = pd.Series(predicted_labels).value_counts().sort_index()

    print(f"True label counts: {true_count}")
    print(f"Predicted label counts: {pred_count}")

    if not true_count.empty and not pred_count.empty:
        true_count.plot(kind='bar', alpha=0.5, color='blue', position=0, label='True Labels')
        pred_count.plot(kind='bar', alpha=0.5, color='red', position=1, label='Predicted Labels')

        plt.xlabel('Labels')
        plt.ylabel('Count')
        plt.title('True vs Predicted Labels')
        plt.legend()
        plt.show()
    else:
        print("No data to plot.")

    # Add new persons to the gallery if they are not present
    for person_name, embeds in probe_embeddings.items():
        person_name = str(person_name)
        if person_name not in person_to_idx:
            for embed in embeds:
                gallery_embeddings = add_new_person_to_gallery(person_name, embed, loaded_gallery_embeddings)

    # Save the updated gallery to JSON
    save_gallery_to_json(loaded_gallery_embeddings, gallery_json_path)
    print(f"Updated gallery saved to {gallery_json_path}")

This code snippet provides functions to handle embeddings, match new individuals with a gallery, and evaluate the model. Here’s a concise explanation of each function:¶

create_embedding_for_new_person:

Creates embeddings for new data using the model. Sets the model to evaluation mode and generates person features without gradient calculation.

match_new_person_with_gallery:

Matches new embeddings with existing gallery embeddings using cosine similarity. Computes similarity scores between the new embedding and each gallery embedding.

add_new_person_to_gallery:

Adds a new person's embedding to the gallery. Appends the new embedding to the list of embeddings for that person.

evaluate_model:

Evaluates the model on a validation dataset.Loads gallery embeddings and creates embeddings for the validation set.Calculates cosine similarity between gallery and validation embeddings.Maps person names to indices and calculates accuracy.Plots a confusion matrix and bar charts for true vs. predicted labels. Adds new individuals to the gallery and saves the updated gallery to a JSON file.

def evaluate_model_and_predict(model, val_dataset, gallery_json_path, device, similarity_threshold=0.5):
    # Load gallery embeddings from JSON file
    loaded_gallery_embeddings = read_gallery_from_json(gallery_json_path)
    print(f"Loaded gallery embeddings: {len(loaded_gallery_embeddings)} persons")

    # Create embeddings for the probe set (validation set)
    model.eval()
    probe_embeddings = {}
    style_embeddings = {}
    probe_labels = []
    style_labels = []
    with torch.no_grad():
        for images, person_labels, style_labels_batch in DataLoader(val_dataset, batch_size=32, shuffle=False):
            images = images.to(device)
            person_labels = person_labels.to(device)
            style_labels_batch = style_labels_batch.to(device)
            person_output, style_output ,embeddings= model(images)
            for embedding, person_label, style_label in zip(embeddings, person_labels, style_labels_batch):
                person_name = val_dataset.index_to_person[person_label.item()]
                style_name = val_dataset.index_to_style[style_label.item()]
                if person_name not in probe_embeddings:
                    probe_embeddings[person_name] = []
                if style_name not in style_embeddings:
                    style_embeddings[style_name] = []
                probe_embeddings[person_name].append(embedding.cpu())
                style_embeddings[style_name].append(embedding.cpu())
                probe_labels.append(person_name)
                style_labels.append(style_name)

    print(f"Created probe embeddings: {len(probe_embeddings)} persons")
    print(f"Created style embeddings: {len(style_embeddings)} styles")
    total_val_images = sum(len(embeds) for embeds in probe_embeddings.values())
    print(f"Total validation images processed: {total_val_images}")

    # Identity evaluation
    all_probe_embeddings = []
    all_probe_labels = []
    for person_name, embeddings in probe_embeddings.items():
        all_probe_embeddings.extend(embeddings)
        all_probe_labels.extend([person_name] * len(embeddings))

    probe_tensors = torch.stack(all_probe_embeddings)
    gallery_tensors = torch.stack([torch.mean(torch.stack(embeds), dim=0) for embeds in loaded_gallery_embeddings.values()])

    person_to_idx = {person: idx for idx, person in enumerate(loaded_gallery_embeddings.keys())}
    idx_to_person = {idx: person for person, idx in person_to_idx.items()}
    unknown_idx = len(person_to_idx)

from torch.utils.data import DataLoader
from helper import read_gallery_from_json, write_predictions_to_json, encode_embedding

def evaluate_test_set_write(model, test_dataset, gallery_json_path, device, output_json_path, similarity_threshold=0.5):
    """
    Evaluate the model on the test set and write predictions to a JSON file.

    Args:
    - model: The trained model to be evaluated.
    - test_dataset: The test dataset.
    - gallery_json_path: Path to the gallery embeddings JSON file.
    - device: Device to run the model on (e.g., 'cuda' or 'cpu').
    - output_json_path: Path to the output JSON file for predictions.
    - similarity_threshold: Threshold for similarity to consider a match.

    Returns:
    - predictions: List of tuples containing filename, embedding, and style.
    """
    # Load gallery embeddings from JSON file
    loaded_gallery_embeddings = read_gallery_from_json(gallery_json_path)
    print(f"Loaded gallery embeddings: {len(loaded_gallery_embeddings)} persons")

    # Create embeddings for the probe set (test set)
    model.eval()
    probe_embeddings = []
    filenames = []
    style_outputs_list = []

    with torch.no_grad():
        for images, paths in DataLoader(test_dataset, batch_size=32, shuffle=False):
            images = images.to(device)
            _, style_outputs,embeddings = model(images)
            for embedding, path, style_output in zip(embeddings, paths, style_outputs):
                probe_embeddings.append(embedding.cpu())
                filenames.append(path)
                style_outputs_list.append(style_output.cpu())

    print(f"Created probe embeddings for {len(filenames)} images")

    # Generate predictions
    predictions = []
    probe_tensors = torch.stack(probe_embeddings)
    gallery_tensors = torch.stack([torch.mean(torch.stack(embeds), dim=0) for embeds in loaded_gallery_embeddings.values()])

    cos_sim = torch.matmul(probe_tensors, gallery_tensors.T)
    cos_sim = cos_sim / (torch.norm(probe_tensors, dim=1, keepdim=True) * torch.norm(gallery_tensors, dim=1))

    person_to_idx = {person: idx for idx, person in enumerate(loaded_gallery_embeddings.keys())}
    idx_to_person = {idx: person for person, idx in person_to_idx.items()}
    unknown_idx = len(person_to_idx)
    idx_to_person[unknown_idx] = "unknown"

    for i, similarities in enumerate(cos_sim):
        max_similarity = torch.max(similarities).item()
        if max_similarity < similarity_threshold:
            predicted_person = "unknown"
        else:
            predicted_person = idx_to_person[torch.argmax(similarities).item()]
        predicted_style = torch.argmax(style_outputs_list[i]).item()  # Predicting style

        predictions.append((filenames[i], probe_tensors[i], predicted_style))

    write_predictions_to_json(predictions, output_json_path)

    return predictions
